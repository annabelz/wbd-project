# *SPARQL QUERY USED TO CREATE DiseaseMeSH.csv*

SELECT DISTINCT ?itemLabel ?meshID ?treeCode ?icd10 WHERE {
  # MeSH tree code
  ?item wdt:P672 ?treeCode .
  FILTER(STRSTARTS(?treeCode, "F03.625.164"))

  # MeSH descriptor ID
  ?item wdt:P486 ?meshID .

  # OPTIONAL ICD-10 code
  OPTIONAL { ?item wdt:P494 ?icd10 . }

  SERVICE wikibase:label {
    bd:serviceParam wikibase:language "en" .
  }
}
ORDER BY ?treeCode
